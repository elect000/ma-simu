function [] = createGraph(xfrom, xto, m, a)
x = linspace(xfrom, xto, m);
y = zeros(1, m);
n = size(a, 2) - 1;
for i=1:n
    y = (y + a(i)) .* x;
end
y = y + a(n + 1);
plot(x, y)
endfunction
