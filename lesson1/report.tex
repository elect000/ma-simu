   2.2  -0.2  -0.2
\end{verbatim}

function [val1, val2] = myfunc(vec)
[vals, idxs] = gsort(abs(vec))
idxs = flipdim(idxs, 2)
val1 = vec(idxs(1))
val2 =  vec(idxs(2))
endfunction
\end{verbatim}
