x = repmat(x, n, 1);
v = x
e = 1;
e = repmat(e, n, 1);

  R(count + 1) = 1/norm(x) * norm((alpha * P + (1 - alpha) * v * e') * x - x);
  x = alpha * P * x + (1 - alpha) * v;

end

endfunction
